export function AddonUrl({ addon }) {
  return window.location.origin + useBaseUrl(`/stremio/addons?addon=${addon}`);
}

# Stremio Addons

<Tabs queryString="addon" className="hidden-tabs">
<TabItem default value="">
This page contains all the instructions for each addon covered in the [setup](setup#setting-up-addons) section.

To view the instructions for a specific addon, append `?addon=addon-name` to the URL, where `addon-name` is the name of the addon you want to view.

For example, to view the instructions for Streaming Catalogs, you would visit the following URL: 

<a href={AddonUrl({ addon: 'streaming-catalogs'})}>{AddonUrl({ addon: 'streaming-catalogs'})}</a><br/><br/>
